\newcommand{\centeredFrame}[1]{
  \frame{
    \frametitle{\insertsection}
    \begin{center}
      {#1}
    \end{center}
  }
}

\newcommand{\itemizedFrame}[2]{
  \frame{
    \frametitle{\insertsection}
    {#1}
    \begin{itemize}
      {#2}
    \end{itemize}
  }
}

\newcommand{\imageFrame}[1]{
  \frame[plain]{
    \begin{changemargin}{-1cm}{-1cm}
      \begin{center}
        \includegraphics[width=\paperwidth,height=\paperheight,keepaspectratio]{#1}
      \end{center}
    \end{changemargin}
  }
}

\newcommand{\inputFrame}[3]{
  \frame{
    \frametitle{{#1}}
    {#2} \input {#3} % FIXME formatting {#2} doesn't work anymore
  }
}
